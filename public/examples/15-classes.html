class Persoon {
    constructor(voornaam, achternaam) {
        this.voornaam = voornaam;
        this.achternaam = achternaam;
        if(!Persoon.personen) {
            Persoon.personen = [];
        }
        Persoon.personen.push(this);
    }
    
    get volledigeNaam() {
        return this.voornaam + ' ' + this.achternaam;
    }
    
    static allePersonen() {
        const personen = Persoon.personen.map(function (persoon) {
            return persoon.volledigeNaam;
        });
        return personen.join(', ');
    }
}

class Waifu extends Persoon {
    zetWiensWaifu(persoon) {
        this.persoon = persoon;
    }
}

class JapaneseWaifu extends Waifu {
    get volledigeNaam() {
        return this.achternaam + ' ' + this.voornaam;
    }
}

class BNer extends Persoon {
    constructor(voornaam, achternaam, beroep) {
        super(voornaam, achternaam);
        this.beroep = beroep;
    }

    get volledigeNaam() {
        return super.volledigeNaam + ' (' + this.beroep + ')';
    }
}
